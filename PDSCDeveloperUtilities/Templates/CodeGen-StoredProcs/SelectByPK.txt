
<|COMMA_RESET|>
CREATE OR ALTER PROCEDURE [<|SCHEMA_NAME|>].[<|SP_PREFIX|><|TABLE_NAME|>SelectByPK]
{|FOR EACH COLUMN:IsPrimaryKey|}
<|COMMA|>@<|PROPERTY_NAME|> <|DATA_TYPE|>
{|END_LOOP|}
AS
SELECT 
<|COMMA_RESET|>
{|FOR EACH COLUMN|}
  <|COMMA|>[<|COLUMN_NAME|>]
{|END_LOOP|}
FROM [<|SCHEMA_NAME|>].[<|TABLE_NAME|>]
<|AND_RESET|>
WHERE
{|FOR EACH COLUMN:IsPrimaryKey|}
  <|AND_UPPER|> [<|COLUMN_NAME|>] = @<|PROPERTY_NAME|>
{|END_LOOP|}
GO
