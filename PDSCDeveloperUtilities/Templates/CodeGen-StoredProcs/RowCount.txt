
<|COMMA_RESET|>
CREATE OR ALTER PROCEDURE [<|SCHEMA_NAME|>].[<|SP_PREFIX|><|TABLE_NAME|>Count]
{|FOR EACH COLUMN:IsSearchField|}
<|COMMA|>@<|PROPERTY_NAME|> <|DATA_TYPE|> = null
{|END_LOOP|}
AS
SELECT COUNT(*) As RecordsAffected
FROM [<|SCHEMA_NAME|>].[<|TABLE_NAME|>]
<|AND_RESET|>
WHERE
{|FOR EACH COLUMN:IsSearchField|}
  <|AND_UPPER|> (@<|PROPERTY_NAME|> IS NULL OR [<|COLUMN_NAME|>] LIKE @<|PROPERTY_NAME|> + '%')
{|END_LOOP|}
GO
